296       # Instantiates a new column for the table.
297       # The +type+ parameter is normally one of the migrations native types,
298       # which is one of the following:
299       # <tt>:primary_key</tt>, <tt>:string</tt>, <tt>:text</tt>,
300       # <tt>:integer</tt>, <tt>:float</tt>, <tt>:decimal</tt>,
301       # <tt>:datetime</tt>, <tt>:timestamp</tt>, <tt>:time</tt>,
302       # <tt>:date</tt>, <tt>:binary</tt>, <tt>:boolean</tt>.
303       #
304       # You may use a type not in this list as long as it is supported by your
305       # database (for example, "polygon" in MySQL), but this will not be database
306       # agnostic and should usually be avoided.
307       #
308       # Available options are (none of these exists by default):
309       # * <tt>:limit</tt> -
310       #   Requests a maximum column length. This is number of characters for <tt>:string</tt> and <tt>:text</tt> columns and number of bytes for :binary and :integer columns.
311       # * <tt>:default</tt> -
312       #   The column's default value. Use nil for NULL.
313       # * <tt>:null</tt> -
314       #   Allows or disallows +NULL+ values in the column. This option could
315       #   have been named <tt>:null_allowed</tt>.
316       # * <tt>:precision</tt> -
317       #   Specifies the precision for a <tt>:decimal</tt> column.
318       # * <tt>:scale</tt> -
319       #   Specifies the scale for a <tt>:decimal</tt> column.
320       #
321       # Please be aware of different RDBMS implementations behavior with
322       # <tt>:decimal</tt> columns:
323       # * The SQL standard says the default scale should be 0, <tt>:scale</tt> <=
324       #   <tt>:precision</tt>, and makes no comments about the requirements of
325       #   <tt>:precision</tt>.
326       # * MySQL: <tt>:precision</tt> [1..63], <tt>:scale</tt> [0..30].
327       #   Default is (10,0).
328       # * PostgreSQL: <tt>:precision</tt> [1..infinity],
329       #   <tt>:scale</tt> [0..infinity]. No default.
330       # * SQLite2: Any <tt>:precision</tt> and <tt>:scale</tt> may be used.
331       #   Internal storage as strings. No default.
332       # * SQLite3: No restrictions on <tt>:precision</tt> and <tt>:scale</tt>,
333       #   but the maximum supported <tt>:precision</tt> is 16. No default.
334       # * Oracle: <tt>:precision</tt> [1..38], <tt>:scale</tt> [-84..127].
335       #   Default is (38,0).
336       # * DB2: <tt>:precision</tt> [1..63], <tt>:scale</tt> [0..62].
337       #   Default unknown.
338       # * Firebird: <tt>:precision</tt> [1..18], <tt>:scale</tt> [0..18].
339       #   Default (9,0). Internal types NUMERIC and DECIMAL have different
340       #   storage rules, decimal being better.
341       # * FrontBase?: <tt>:precision</tt> [1..38], <tt>:scale</tt> [0..38].
342       #   Default (38,0). WARNING Max <tt>:precision</tt>/<tt>:scale</tt> for
343       #   NUMERIC is 19, and DECIMAL is 38.
344       # * SqlServer?: <tt>:precision</tt> [1..38], <tt>:scale</tt> [0..38].
345       #   Default (38,0).
346       # * Sybase: <tt>:precision</tt> [1..38], <tt>:scale</tt> [0..38].

374       # == Short-hand examples
375       #
376       # Instead of calling +column+ directly, you can also work with the short-hand definitions for the default types.
377       # They use the type as the method name instead of as a parameter and allow for multiple columns to be defined
378       # in a single statement.
379       #
380       # What can be written like this with the regular calls to column:
381       #
382       #   create_table "products", :force => true do |t|
383       #     t.column "shop_id",    :integer
384       #     t.column "creator_id", :integer
385       #     t.column "name",       :string,   :default => "Untitled"
386       #     t.column "value",      :string,   :default => "Untitled"
387       #     t.column "created_at", :datetime
388       #     t.column "updated_at", :datetime
389       #   end
390       #
391       # Can also be written as follows using the short-hand:
392       #
393       #   create_table :products do |t|
394       #     t.integer :shop_id, :creator_id
395       #     t.string  :name, :value, :default => "Untitled"
396       #     t.timestamps
397       #   end
398       #
399       # There's a short-hand method for each of the type values declared at the top. And then there's
400       # TableDefinition#timestamps that'll add created_at and +updated_at+ as datetimes.
401       #
402       # TableDefinition#references will add an appropriately-named _id column, plus a corresponding _type
403       # column if the <tt>:polymorphic</tt> option is supplied. If <tt>:polymorphic</tt> is a hash of options, these will be
404       # used when creating the <tt>_type</tt> column. So what can be written like this:
405       #
406       #   create_table :taggings do |t|
407       #     t.integer :tag_id, :tagger_id, :taggable_id
408       #     t.string  :tagger_type
409       #     t.string  :taggable_type, :default => 'Photo'
410       #   end
411       #
412       # Can also be written as follows using references:
413       #
414       #   create_table :taggings do |t|
415       #     t.references :tag
416       #     t.references :tagger, :polymorphic => true
417       #     t.references :taggable, :polymorphic => { :default => 'Photo' }
418       #   end

420         column = self[name] || ColumnDefinition.new(@base, name, type)
421         if options[:limit]
422           column.limit = options[:limit]
423         elsif native[type.to_sym].is_a?(Hash)

427         column.scale = options[:scale]
428         column.default = options[:default]
429         column.null = options[:null]
