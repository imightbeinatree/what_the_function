400       def assert_select_rjs(*args, &block)
401         rjs_type = args.first.is_a?(Symbol) ? args.shift : nil
402         id       = args.first.is_a?(String) ? args.shift : nil
403 
404         # If the first argument is a symbol, it's the type of RJS statement we're looking
405         # for (update, replace, insertion, etc). Otherwise, we're looking for just about
406         # any RJS statement.
407         if rjs_type
408           if rjs_type == :insert
409             position  = args.shift
410             insertion = "insert_#{position}".to_sym
411             raise ArgumentError, "Unknown RJS insertion type #{position}" unless RJS_STATEMENTS[insertion]
412             statement = "(#{RJS_STATEMENTS[insertion]})"
413           else
414             raise ArgumentError, "Unknown RJS statement type #{rjs_type}" unless RJS_STATEMENTS[rjs_type]
415             statement = "(#{RJS_STATEMENTS[rjs_type]})"
416           end
417         else
418           statement = "#{RJS_STATEMENTS[:any]}"
419         end
420 
421         # Next argument we're looking for is the element identifier. If missing, we pick
422         # any element, otherwise we replace it in the statement.
423         pattern = Regexp.new(
424           id ? statement.gsub(RJS_ANY_ID, "\"#{id}\"") : statement
425         )
426 
427         # Duplicate the body since the next step involves destroying it.
428         matches = nil
429         case rjs_type
430           when :remove, :show, :hide, :toggle
431             matches = @response.body.match(pattern)
432           else
433             @response.body.gsub(pattern) do |match|
434               html = unescape_rjs(match)
435               matches ||= []
436               matches.concat HTML::Document.new(html).root.children.select { |n| n.tag? }
437               ""
438             end
439         end

441         if matches
442           assert_block("") { true } # to count the assertion
443           if block_given? && !([:remove, :show, :hide, :toggle].include? rjs_type)
444             begin
445               in_scope, @selected = @selected, matches

447             ensure
448               @selected = in_scope
449             end
450           end

452         else
453           # RJS statement not found.

609         # Unescapes a RJS string.
610         def unescape_rjs(rjs_string)
611           # RJS encodes double quotes and line breaks.
612           unescaped= rjs_string.gsub('\"', '"')
613           unescaped.gsub!(/\\\//, '/')
614           unescaped.gsub!('\n', "\n")
615           unescaped.gsub!('\076', '>')
616           unescaped.gsub!('\074', '<')
617           # RJS encodes non-ascii characters.
