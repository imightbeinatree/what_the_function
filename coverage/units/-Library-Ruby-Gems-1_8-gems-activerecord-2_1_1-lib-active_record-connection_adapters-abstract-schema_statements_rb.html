229       # Adds a new index to the table.  +column_name+ can be a single Symbol, or
230       # an Array of Symbols.
231       #
232       # The index will be named after the table and the first column name,
233       # unless you pass <tt>:name</tt> as an option.
234       #
235       # When creating an index on multiple columns, the first column is used as a name
236       # for the index. For example, when you specify an index on two columns
237       # [<tt>:first</tt>, <tt>:last</tt>], the DBMS creates an index for both columns as well as an
238       # index for the first column <tt>:first</tt>. Using just the first name for this index
239       # makes sense, because you will never have to create a singular index with this
240       # name.
241       #
242       # ===== Examples
243       # ====== Creating a simple index
244       #  add_index(:suppliers, :name)
245       # generates
246       #  CREATE INDEX suppliers_name_index ON suppliers(name)
247       # ====== Creating a unique index
248       #  add_index(:accounts, [:branch_id, :party_id], :unique => true)
249       # generates
250       #  CREATE UNIQUE INDEX accounts_branch_id_party_id_index ON accounts(branch_id, party_id)
251       # ====== Creating a named index
252       #  add_index(:accounts, [:branch_id, :party_id], :unique => true, :name => 'by_branch_party')
253       # generates
254       #  CREATE UNIQUE INDEX by_branch_party ON accounts(branch_id, party_id)
255       def add_index(table_name, column_name, options = {})
256         column_names = Array(column_name)
257         index_name   = index_name(table_name, :column => column_names)
258 
259         if Hash === options # legacy support, since this param was a string
260           index_type = options[:unique] ? "UNIQUE" : ""
261           index_name = options[:name] || index_name
262         else
263           index_type = options
264         end
265         quoted_column_names = column_names.map { |e| quote_column_name(e) }.join(", ")
266         execute "CREATE #{index_type} INDEX #{quote_column_name(index_name)} ON #{quote_table_name(table_name)} (#{quoted_column_names})"
267       end

333       def assume_migrated_upto_version(version)
334         version = version.to_i
335         sm_table = quote_table_name(ActiveRecord::Migrator.schema_migrations_table_name)
336 
337         migrated = select_values("SELECT version FROM #{sm_table}").map(&:to_i)
338         versions = Dir['db/migrate/[0-9]*_*.rb'].map do |filename|
339           filename.split('/').last.split('_').first.to_i
340         end
341 
342         unless migrated.include?(version)
343           execute "INSERT INTO #{sm_table} (version) VALUES ('#{version}')"
344         end
345 
346         inserted = Set.new
347         (versions - migrated).each do |v|
348           if inserted.include?(v)
349             raise "Duplicate migration #{v}. Please renumber your migrations to resolve the conflict."
350           elsif v < version
351             execute "INSERT INTO #{sm_table} (version) VALUES ('#{v}')"
352             inserted << v
353           end
354         end

357       def type_to_sql(type, limit = nil, precision = nil, scale = nil) #:nodoc:
358         if native = native_database_types[type]
359           column_type_sql = native.is_a?(Hash) ? native[:name] : native
360 
361           if type == :decimal # ignore limit, use precision and scale
362             scale ||= native[:scale]
363 
364             if precision ||= native[:precision]
365               if scale
366                 column_type_sql << "(#{precision},#{scale})"
367               else
368                 column_type_sql << "(#{precision})"
369               end
370             elsif scale
371               raise ArgumentError, "Error adding decimal column: precision cannot be empty if scale if specified"
372             end
373 
374           elsif limit ||= native.is_a?(Hash) && native[:limit]
375             column_type_sql << "(#{limit})"
376           end
377 
378           column_type_sql
379         else
380           type
381         end
382       end
