 12       # Returns the URL for the set of +options+ provided. This takes the
 13       # same options as +url_for+ in Action Controller (see the
 14       # documentation for ActionController::Base#url_for). Note that by default
 15       # <tt>:only_path</tt> is <tt>true</tt> so you'll get the relative /controller/action
 16       # instead of the fully qualified URL like http://example.com/controller/action.
 17       #
 18       # When called from a view, url_for returns an HTML escaped url. If you
 19       # need an unescaped url, pass <tt>:escape => false</tt> in the +options+.
 20       #
 21       # ==== Options
 22       # * <tt>:anchor</tt> - Specifies the anchor name to be appended to the path.
 23       # * <tt>:only_path</tt> - If true, returns the relative URL (omitting the protocol, host name, and port) (<tt>true</tt> by default unless <tt>:host</tt> is specified).
 24       # * <tt>:trailing_slash</tt> - If true, adds a trailing slash, as in "/archive/2005/". Note that this
 25       #   is currently not recommended since it breaks caching.
 26       # * <tt>:host</tt> - Overrides the default (current) host if provided.
 27       # * <tt>:protocol</tt> - Overrides the default (current) protocol if provided.
 28       # * <tt>:user</tt> - Inline HTTP authentication (only plucked out if <tt>:password</tt> is also present).
 29       # * <tt>:password</tt> - Inline HTTP authentication (only plucked out if <tt>:user</tt> is also present).
 30       # * <tt>:escape</tt> - Determines whether the returned URL will be HTML escaped or not (<tt>true</tt> by default).
 31       #
 32       # ==== Relying on named routes
 33       #
 34       # If you instead of a hash pass a record (like an Active Record or Active Resource) as the options parameter,
 35       # you'll trigger the named route for that record. The lookup will happen on the name of the class. So passing
 36       # a Workshop object will attempt to use the workshop_path route. If you have a nested route, such as 
 37       # admin_workshop_path you'll have to call that explicitly (it's impossible for url_for to guess that route). 
 38       #
 39       # ==== Examples
 40       #   <%= url_for(:action => 'index') %>
 41       #   # => /blog/
 42       #
 43       #   <%= url_for(:action => 'find', :controller => 'books') %>
 44       #   # => /books/find
 45       #
 46       #   <%= url_for(:action => 'login', :controller => 'members', :only_path => false, :protocol => 'https') %>
 47       #   # => https://www.railsapplication.com/members/login/
 48       #
 49       #   <%= url_for(:action => 'play', :anchor => 'player') %>
 50       #   # => /messages/play/#player
 51       #
 52       #   <%= url_for(:action => 'checkout', :anchor => 'tax&ship') %>
 53       #   # => /testing/jump/#tax&amp;ship
 54       #
 55       #   <%= url_for(:action => 'checkout', :anchor => 'tax&ship', :escape => false) %>
 56       #   # => /testing/jump/#tax&ship
 57       #
 58       #   <%= url_for(Workshop.new) %>
 59       #   # relies on Workshop answering a new_record? call (and in this case returning true)
 60       #   # => /workshops
 61       #
 62       #   <%= url_for(@workshop) %>
 63       #   # calls @workshop.to_s
 64       #   # => /workshops/5
 65       def url_for(options = {})
 66         options ||= {}
 67         case options
 68         when Hash
 69           options = { :only_path => options[:host].nil? }.update(options.symbolize_keys)
 70           escape  = options.key?(:escape) ? options.delete(:escape) : true
 71           url     = @controller.send(:url_for, options)
 72         when String
 73           escape = true
 74           url    = options
 75         else
 76           escape = false
 77           url    = polymorphic_path(options)
 78         end
 79 
 80         escape ? escape_once(url) : url
 81       end

383       # Creates a mailto link tag to the specified +email_address+, which is
384       # also used as the name of the link unless +name+ is specified. Additional
385       # HTML attributes for the link can be passed in +html_options+.
386       #
387       # mail_to has several methods for hindering email harvesters and customizing
388       # the email itself by passing special keys to +html_options+.
389       #
390       # ==== Options
391       # * <tt>:encode</tt>  - This key will accept the strings "javascript" or "hex".
392       #   Passing "javascript" will dynamically create and encode the mailto: link then
393       #   eval it into the DOM of the page. This method will not show the link on
394       #   the page if the user has JavaScript disabled. Passing "hex" will hex
395       #   encode the +email_address+ before outputting the mailto: link.
396       # * <tt>:replace_at</tt>  - When the link +name+ isn't provided, the
397       #   +email_address+ is used for the link label. You can use this option to
398       #   obfuscate the +email_address+ by substituting the @ sign with the string
399       #   given as the value.
400       # * <tt>:replace_dot</tt>  - When the link +name+ isn't provided, the
401       #   +email_address+ is used for the link label. You can use this option to
402       #   obfuscate the +email_address+ by substituting the . in the email with the
403       #   string given as the value.
404       # * <tt>:subject</tt>  - Preset the subject line of the email.
405       # * <tt>:body</tt> - Preset the body of the email.
406       # * <tt>:cc</tt>  - Carbon Copy addition recipients on the email.
407       # * <tt>:bcc</tt>  - Blind Carbon Copy additional recipients on the email.
408       #
409       # ==== Examples
410       #   mail_to "me@domain.com" 
411       #   # => <a href="mailto:me@domain.com">me@domain.com</a>
412       #
413       #   mail_to "me@domain.com", "My email", :encode => "javascript"  
414       #   # => <script type="text/javascript">eval(unescape('%64%6f%63...%6d%65%6e'))</script>
415       #
416       #   mail_to "me@domain.com", "My email", :encode => "hex"  
417       #   # => <a href="mailto:%6d%65@%64%6f%6d%61%69%6e.%63%6f%6d">My email</a>
418       #
419       #   mail_to "me@domain.com", nil, :replace_at => "_at_", :replace_dot => "_dot_", :class => "email"  
420       #   # => <a href="mailto:me@domain.com" class="email">me_at_domain_dot_com</a>
421       #
422       #   mail_to "me@domain.com", "My email", :cc => "ccaddress@domain.com",
423       #            :subject => "This is an example email"  
424       #   # => <a href="mailto:me@domain.com?cc=ccaddress@domain.com&subject=This%20is%20an%20example%20email">My email</a>
425       def mail_to(email_address, name = nil, html_options = {})
426         html_options = html_options.stringify_keys
427         encode = html_options.delete("encode").to_s
428         cc, bcc, subject, body = html_options.delete("cc"), html_options.delete("bcc"), html_options.delete("subject"), html_options.delete("body")
429 
430         string = ''
431         extras = ''
432         extras << "cc=#{CGI.escape(cc).gsub("+", "%20")}&" unless cc.nil?
433         extras << "bcc=#{CGI.escape(bcc).gsub("+", "%20")}&" unless bcc.nil?
434         extras << "body=#{CGI.escape(body).gsub("+", "%20")}&" unless body.nil?
435         extras << "subject=#{CGI.escape(subject).gsub("+", "%20")}&" unless subject.nil?
436         extras = "?" << extras.gsub!(/&?$/,"") unless extras.empty?
437 
438         email_address = email_address.to_s
439 
440         email_address_obfuscated = email_address.dup
441         email_address_obfuscated.gsub!(/@/, html_options.delete("replace_at")) if html_options.has_key?("replace_at")
442         email_address_obfuscated.gsub!(/\./, html_options.delete("replace_dot")) if html_options.has_key?("replace_dot")
443 
444         if encode == "javascript"
445           "document.write('#{content_tag("a", name || email_address_obfuscated, html_options.merge({ "href" => "mailto:"+email_address+extras }))}');".each_byte do |c|
446             string << sprintf("%%%x", c)
447           end
448           "<script type=\"#{Mime::JS}\">eval(unescape('#{string}'))</script>"
449         elsif encode == "hex"
450           email_address_encoded = ''
451           email_address_obfuscated.each_byte do |c|
452             email_address_encoded << sprintf("&#%d;", c)
453           end
454 
455           protocol = 'mailto:'
456           protocol.each_byte { |c| string << sprintf("&#%d;", c) }
457 
458           email_address.each_byte do |c|
459             char = c.chr
460             string << (char =~ /\w/ ? sprintf("%%%x", c) : char)
461           end
462           content_tag "a", name || email_address_encoded, html_options.merge({ "href" => "#{string}#{extras}" })
463         else
464           content_tag "a", name || email_address_obfuscated, html_options.merge({ "href" => "mailto:#{email_address}#{extras}" })
465         end
466       end

470       # ==== Examples
471       # Let's say we're in the <tt>/shop/checkout</tt> action.
472       #
473       #   current_page?(:action => 'process')
474       #   # => false
475       #
476       #   current_page?(:controller => 'shop', :action => 'checkout')
477       #   # => true
478       #
479       #   current_page?(:action => 'checkout')
480       #   # => true
481       #
482       #   current_page?(:controller => 'library', :action => 'checkout')
483       #   # => false
484       def current_page?(options)
485         url_string = CGI.escapeHTML(url_for(options))
486         request = @controller.request
487         if url_string =~ /^\w+:\/\//
488           url_string == "#{request.protocol}#{request.host_with_port}#{request.request_uri}"
489         else
490           url_string == request.request_uri
491         end
492       end

495         def convert_options_to_javascript!(html_options, url = '')
496           confirm, popup = html_options.delete("confirm"), html_options.delete("popup")
497 
498           method, href = html_options.delete("method"), html_options['href']
499 
500           html_options["onclick"] = case
501             when popup && method
502               raise ActionView::ActionViewError, "You can't use :popup and :method in the same link"
503             when confirm && popup
504               "if (#{confirm_javascript_function(confirm)}) { #{popup_javascript_function(popup)} };return false;"
505             when confirm && method
506               "if (#{confirm_javascript_function(confirm)}) { #{method_javascript_function(method)} };return false;"
507             when confirm
508               "return #{confirm_javascript_function(confirm)};"
509             when method
510               "#{method_javascript_function(method, url, href)}return false;"
511             when popup
512               popup_javascript_function(popup) + 'return false;'
513             else
514               html_options["onclick"]
515           end
516         end

520         end

522         def popup_javascript_function(popup)
523           popup.is_a?(Array) ? "window.open(this.href,'#{popup.first}','#{popup.last}');" : "window.open(this.href);"
524         end

526         def method_javascript_function(method, url = '', href = nil)
527           action = (href && url.size > 0) ? "'#{url}'" : 'this.href'
528           submit_function =
529             "var f = document.createElement('form'); f.style.display = 'none'; " +
530             "this.parentNode.appendChild(f); f.method = 'POST'; f.action = #{action};"
531 
532           unless method == :post
533             submit_function << "var m = document.createElement('input'); m.setAttribute('type', 'hidden'); "
534             submit_function << "m.setAttribute('name', '_method'); m.setAttribute('value', '#{method}'); f.appendChild(m);"
535           end
536 
537           if protect_against_forgery?
538             submit_function << "var s = document.createElement('input'); s.setAttribute('type', 'hidden'); "
539             submit_function << "s.setAttribute('name', '#{request_forgery_protection_token}'); s.setAttribute('value', '#{escape_javascript form_authenticity_token}'); f.appendChild(s);"
540           end
541           submit_function << "f.submit();"
542         end

544         # Processes the _html_options_ hash, converting the boolean
545         # attributes from true/false form into the form required by
546         # HTML/XHTML.  (An attribute is considered to be boolean if
547         # its name is listed in the given _bool_attrs_ array.)
548         #
549         # More specifically, for each boolean attribute in _html_options_
550         # given as:
551         #
552         #     "attr" => bool_value
553         #
554         # if the associated _bool_value_ evaluates to true, it is
555         # replaced with the attribute's name; otherwise the attribute is
556         # removed from the _html_options_ hash.  (See the XHTML 1.0 spec,
557         # section 4.5 "Attribute Minimization" for more:
558         # http://www.w3.org/TR/xhtml1/#h-4.5)
559         #
560         # Returns the updated _html_options_ hash, which is also modified
561         # in place.
562         #
563         # Example:
564         #
565         #   convert_boolean_attributes!( html_options,
566         #                                %w( checked disabled readonly ) )
567         def convert_boolean_attributes!(html_options, bool_attrs)
568           bool_attrs.each { |x| html_options[x] = x if html_options.delete(x) }
569           html_options
570         end
