499       # * <tt>:message</tt> - The error message to use for a <tt>:minimum</tt>, <tt>:maximum</tt>, or <tt>:is</tt> violation.  An alias of the appropriate <tt>too_long</tt>/<tt>too_short</tt>/<tt>wrong_length</tt> message.
500       # * <tt>:on</tt> - Specifies when this validation is active (default is <tt>:save</tt>, other options <tt>:create</tt>, <tt>:update</tt>).
501       # * <tt>:if</tt> - Specifies a method, proc or string to call to determine if the validation should
502       #   occur (e.g. <tt>:if => :allow_validation</tt>, or <tt>:if => Proc.new { |user| user.signup_step > 2 }</tt>).  The
503       #   method, proc or string should return or evaluate to a true or false value.
504       # * <tt>:unless</tt> - Specifies a method, proc or string to call to determine if the validation should
505       #   not occur (e.g. <tt>:unless => :skip_validation</tt>, or <tt>:unless => Proc.new { |user| user.signup_step <= 2 }</tt>).  The
506       #   method, proc or string should return or evaluate to a true or false value.
507       # * <tt>:tokenizer</tt> - Specifies how to split up the attribute string. (e.g. <tt>:tokenizer => lambda {|str| str.scan(/\w+/)}</tt> to
508       #   count words as in above example.)
509       #   Defaults to <tt>lambda{ |value| value.split(//) }</tt> which counts individual characters.
510       def validates_length_of(*attrs)
511         # Merge given options with defaults.
512         options = {
513           :too_long     => ActiveRecord::Errors.default_error_messages[:too_long],
514           :too_short    => ActiveRecord::Errors.default_error_messages[:too_short],
515           :wrong_length => ActiveRecord::Errors.default_error_messages[:wrong_length],
516           :tokenizer    => lambda {|value| value.split(//)}
517         }.merge(DEFAULT_VALIDATION_OPTIONS)
518         options.update(attrs.extract_options!.symbolize_keys)
519 
520         # Ensure that one and only one range option is specified.
521         range_options = ALL_RANGE_OPTIONS & options.keys
522         case range_options.size
523           when 0
524             raise ArgumentError, 'Range unspecified.  Specify the :within, :maximum, :minimum, or :is option.'
525           when 1
526             # Valid number of options; do nothing.
527           else
528             raise ArgumentError, 'Too many range options specified.  Choose only one.'
529         end
530 
531         # Get range option and value.
532         option = range_options.first
533         option_value = options[range_options.first]
534 
535         case option
536           when :within, :in
537             raise ArgumentError, ":#{option} must be a Range" unless option_value.is_a?(Range)
538 
539             too_short = options[:too_short] % option_value.begin
540             too_long  = options[:too_long]  % option_value.end
541 
542             validates_each(attrs, options) do |record, attr, value|
543               value = options[:tokenizer].call(value) if value.kind_of?(String)
544               if value.nil? or value.size < option_value.begin
545                 record.errors.add(attr, too_short)
546               elsif value.size > option_value.end
547                 record.errors.add(attr, too_long)
548               end
549             end
550           when :is, :minimum, :maximum
551             raise ArgumentError, ":#{option} must be a nonnegative Integer" unless option_value.is_a?(Integer) and option_value >= 0
552 
553             # Declare different validations per option.
554             validity_checks = { :is => "==", :minimum => ">=", :maximum => "<=" }
555             message_options = { :is => :wrong_length, :minimum => :too_short, :maximum => :too_long }
556 
557             message = (options[:message] || options[message_options[option]]) % option_value
558 
559             validates_each(attrs, options) do |record, attr, value|
560               value = options[:tokenizer].call(value) if value.kind_of?(String)
561               record.errors.add(attr, message) unless !value.nil? and value.size.method(validity_checks[option])[option_value]
562             end
563         end
564       end

569       # Validates whether the value of the specified attributes are unique across the system. Useful for making sure that only one user
570       # can be named "davidhh".
571       #
572       #   class Person < ActiveRecord::Base
573       #     validates_uniqueness_of :user_name, :scope => :account_id
574       #   end
575       #
576       # It can also validate whether the value of the specified attributes are unique based on multiple scope parameters.  For example,
577       # making sure that a teacher can only be on the schedule once per semester for a particular class.
578       #
579       #   class TeacherSchedule < ActiveRecord::Base
580       #     validates_uniqueness_of :teacher_id, :scope => [:semester_id, :class_id]
581       #   end
582       #
583       # When the record is created, a check is performed to make sure that no record exists in the database with the given value for the specified
584       # attribute (that maps to a column). When the record is updated, the same check is made but disregarding the record itself.
585       #
586       # Because this check is performed outside the database there is still a chance that duplicate values
587       # will be inserted in two parallel transactions.  To guarantee against this you should create a
588       # unique index on the field. See +add_index+ for more information.
589       #
590       # Configuration options:
591       # * <tt>:message</tt> - Specifies a custom error message (default is: "has already been taken").
592       # * <tt>:scope</tt> - One or more columns by which to limit the scope of the uniqueness constraint.
593       # * <tt>:case_sensitive</tt> - Looks for an exact match.  Ignored by non-text columns (+false+ by default).
594       # * <tt>:allow_nil</tt> - If set to true, skips this validation if the attribute is +nil+ (default is +false+).
595       # * <tt>:allow_blank</tt> - If set to true, skips this validation if the attribute is blank (default is +false+).
596       # * <tt>:if</tt> - Specifies a method, proc or string to call to determine if the validation should
597       #   occur (e.g. <tt>:if => :allow_validation</tt>, or <tt>:if => Proc.new { |user| user.signup_step > 2 }</tt>).  The
598       #   method, proc or string should return or evaluate to a true or false value.
599       # * <tt>:unless</tt> - Specifies a method, proc or string to call to determine if the validation should
600       #   not occur (e.g. <tt>:unless => :skip_validation</tt>, or <tt>:unless => Proc.new { |user| user.signup_step <= 2 }</tt>).  The
601       #   method, proc or string should return or evaluate to a true or false value.
602       def validates_uniqueness_of(*attr_names)
603         configuration = { :message => ActiveRecord::Errors.default_error_messages[:taken], :case_sensitive => true }
604         configuration.update(attr_names.extract_options!)
605 
606         validates_each(attr_names,configuration) do |record, attr_name, value|
607           # The check for an existing value should be run from a class that
608           # isn't abstract. This means working down from the current class
609           # (self), to the first non-abstract class. Since classes don't know
610           # their subclasses, we have to build the hierarchy between self and
611           # the record's class.
612           class_hierarchy = [record.class]
613           while class_hierarchy.first != self
614             class_hierarchy.insert(0, class_hierarchy.first.superclass)
615           end
616 
617           # Now we can work our way down the tree to the first non-abstract
618           # class (which has a database table to query from).
619           finder_class = class_hierarchy.detect { |klass| !klass.abstract_class? }
620 
621           is_text_column = finder_class.columns_hash[attr_name.to_s].text?
622 
623           if !value.nil? && is_text_column
624             value = value.to_s
625           end
626 
627           if value.nil? || (configuration[:case_sensitive] || !is_text_column)
628             condition_sql = "#{record.class.quoted_table_name}.#{attr_name} #{attribute_condition(value)}"
629             condition_params = [value]
630           else
631             # sqlite has case sensitive SELECT query, while MySQL/Postgresql don't.
632             # Hence, this is needed only for sqlite.
633             condition_sql = "LOWER(#{record.class.quoted_table_name}.#{attr_name}) #{attribute_condition(value)}"
634             condition_params = [value.chars.downcase]
635           end
636 
637           if scope = configuration[:scope]
638             Array(scope).map do |scope_item|
639               scope_value = record.send(scope_item)
640               condition_sql << " AND #{record.class.quoted_table_name}.#{scope_item} #{attribute_condition(scope_value)}"
641               condition_params << scope_value
642             end
643           end
644 
645           unless record.new_record?
646             condition_sql << " AND #{record.class.quoted_table_name}.#{record.class.primary_key} <> ?"
647             condition_params << record.send(:id)
648           end
649 
650           results = finder_class.with_exclusive_scope do
651             connection.select_all(
652               construct_finder_sql(
653                 :select     => "#{connection.quote_column_name(attr_name)}",
654                 :from       => "#{finder_class.quoted_table_name}",
655                 :conditions => [condition_sql, *condition_params]
656               )
657             )
658           end
659 
660           unless results.length.zero?
661             found = true
662 
663             # As MySQL/Postgres don't have case sensitive SELECT queries, we try to find duplicate
664             # column in ruby when case sensitive option
665             if configuration[:case_sensitive] && finder_class.columns_hash[attr_name.to_s].text?
666               found = results.any? { |a| a[attr_name.to_s] == value }
667             end
668 
669             record.errors.add(attr_name, configuration[:message]) if found
670           end
671         end
672       end

826       # * <tt>:allow_nil</tt> - Skip validation if attribute is +nil+ (default is +false+). Notice that for fixnum and float columns empty strings are converted to +nil+.
827       # * <tt>:greater_than</tt> - Specifies the value must be greater than the supplied value.

833       # * <tt>:even</tt> - Specifies the value must be an even number.
834       # * <tt>:if</tt> - Specifies a method, proc or string to call to determine if the validation should
835       #   occur (e.g. <tt>:if => :allow_validation</tt>, or <tt>:if => Proc.new { |user| user.signup_step > 2 }</tt>).  The
836       #   method, proc or string should return or evaluate to a true or false value.
837       # * <tt>:unless</tt> - Specifies a method, proc or string to call to determine if the validation should
838       #   not occur (e.g. <tt>:unless => :skip_validation</tt>, or <tt>:unless => Proc.new { |user| user.signup_step <= 2 }</tt>).  The
839       #   method, proc or string should return or evaluate to a true or false value.
840       def validates_numericality_of(*attr_names)
841         configuration = { :on => :save, :only_integer => false, :allow_nil => false }
842         configuration.update(attr_names.extract_options!)
843 
844 
845         numericality_options = ALL_NUMERICALITY_CHECKS.keys & configuration.keys
846 
847         (numericality_options - [ :odd, :even ]).each do |option|
848           raise ArgumentError, ":#{option} must be a number" unless configuration[option].is_a?(Numeric)
849         end
850 
851         validates_each(attr_names,configuration) do |record, attr_name, value|
852           raw_value = record.send("#{attr_name}_before_type_cast") || value
853 
854           next if configuration[:allow_nil] and raw_value.nil?
855 
856           if configuration[:only_integer]
857             unless raw_value.to_s =~ /\A[+-]?\d+\Z/
858               record.errors.add(attr_name, configuration[:message] || ActiveRecord::Errors.default_error_messages[:not_a_number])
859               next
860             end
861             raw_value = raw_value.to_i
862           else
863            begin
864               raw_value = Kernel.Float(raw_value)
865             rescue ArgumentError, TypeError
866               record.errors.add(attr_name, configuration[:message] || ActiveRecord::Errors.default_error_messages[:not_a_number])
867               next
868             end
869           end
870 
871           numericality_options.each do |option|
872             case option
873               when :odd, :even
874                 record.errors.add(attr_name, configuration[:message] || ActiveRecord::Errors.default_error_messages[option]) unless raw_value.to_i.method(ALL_NUMERICALITY_CHECKS[option])[]
875               else
876                 message = configuration[:message] || ActiveRecord::Errors.default_error_messages[option]
877                 message = message % configuration[option] if configuration[option]
878                 record.errors.add(attr_name, message) unless raw_value.method(ALL_NUMERICALITY_CHECKS[option])[configuration[option]]
879             end
880           end
881         end
