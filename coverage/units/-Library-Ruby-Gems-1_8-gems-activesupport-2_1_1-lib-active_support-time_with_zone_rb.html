283       def get_period_and_ensure_valid_local_time
284         # we don't want a Time.local instance enforcing its own DST rules as well, 
285         # so transfer time values to a utc constructor if necessary
286         @time = transfer_time_values_to_utc_constructor(@time) unless @time.utc?
287         begin

289         rescue ::TZInfo::PeriodNotFound

292           retry
293         end
294       end
