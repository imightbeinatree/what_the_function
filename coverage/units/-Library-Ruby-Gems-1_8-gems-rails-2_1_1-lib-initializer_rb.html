882 # Needs to be duplicated from Active Support since its needed before Active
883 # Support is available. Here both Options and Hash are namespaced to prevent
884 # conflicts with other implementations AND with the classes residing in Active Support.
885 class Rails::OrderedOptions < Array #:nodoc:
886   def []=(key, value)
887     key = key.to_sym
888 
889     if pair = find_pair(key)
890       pair.pop
891       pair << value
892     else
893       self << [key, value]
894     end
895   end

897   def [](key)
898     pair = find_pair(key.to_sym)
899     pair ? pair.last : nil
900   end

903     if name.to_s =~ /(.*)=$/
904       self[$1.to_sym] = args.first
