316       # Creates a submit button with the text <tt>value</tt> as the caption. 
317       #
318       # ==== Options
319       # * <tt>:confirm => 'question?'</tt> - This will add a JavaScript confirm
320       #   prompt with the question specified. If the user accepts, the form is
321       #   processed normally, otherwise no action is taken.
322       # * <tt>:disabled</tt> - If true, the user will not be able to use this input.
323       # * <tt>:disable_with</tt> - Value of this parameter will be used as the value for a disabled version 
324       #   of the submit button when the form is submitted.
325       # * Any other key creates standard HTML options for the tag.
326       #
327       # ==== Examples
328       #   submit_tag
329       #   # => <input name="commit" type="submit" value="Save changes" />
330       #
331       #   submit_tag "Edit this article"
332       #   # => <input name="commit" type="submit" value="Edit this article" />
333       #
334       #   submit_tag "Save edits", :disabled => true
335       #   # => <input disabled="disabled" name="commit" type="submit" value="Save edits" />
336       #
337       #   submit_tag "Complete sale", :disable_with => "Please wait..."
338       #   # => <input name="commit" onclick="this.disabled=true;this.value='Please wait...';this.form.submit();" 
339       #   #    type="submit" value="Complete sale" />
340       #
341       #   submit_tag nil, :class => "form_submit"
342       #   # => <input class="form_submit" name="commit" type="submit" />
343       #
344       #   submit_tag "Edit", :disable_with => "Editing...", :class => "edit-button"
345       #   # => <input class="edit-button" onclick="this.disabled=true;this.value='Editing...';this.form.submit();"
346       #   #    name="commit" type="submit" value="Edit" />
347       def submit_tag(value = "Save changes", options = {})
348         options.stringify_keys!
349         
350         if disable_with = options.delete("disable_with")
351           disable_with = "this.value='#{disable_with}'"
352           disable_with << ";#{options.delete('onclick')}" if options['onclick']
353           
354           options["onclick"] = [
355             "this.setAttribute('originalValue', this.value)",
356             "this.disabled=true",
357             disable_with,
358             "result = (this.form.onsubmit ? (this.form.onsubmit() ? this.form.submit() : false) : this.form.submit())",
359             "if (result == false) { this.value = this.getAttribute('originalValue'); this.disabled = false }",
360             "return result;",
361           ].join(";")
362         end
363         
364         if confirm = options.delete("confirm")
365           options["onclick"] ||= ''
366           options["onclick"] += "return #{confirm_javascript_function(confirm)};"
367         end
368         
369         tag :input, { "type" => "submit", "name" => "commit", "value" => value }.update(options.stringify_keys)
370       end

412         concat(tag(:fieldset, {}, true), block.binding)

419         def html_options_for_form(url_for_options, options, *parameters_for_url)
420           returning options.stringify_keys do |html_options|
421             html_options["enctype"] = "multipart/form-data" if html_options.delete("multipart")
422             html_options["action"]  = url_for(url_for_options, *parameters_for_url)
423           end
424         end

426         def extra_tags_for_form(html_options)
427           case method = html_options.delete("method").to_s
428             when /^get$/i # must be case-insentive, but can't use downcase as might be nil
429               html_options["method"] = "get"
430               ''
431             when /^post$/i, "", nil
432               html_options["method"] = "post"
433               protect_against_forgery? ? content_tag(:div, token_tag, :style => 'margin:0;padding:0') : ''
434             else
435               html_options["method"] = "post"
436               content_tag(:div, tag(:input, :type => "hidden", :name => "_method", :value => method) + token_tag, :style => 'margin:0;padding:0')
437           end
438         end
439         
440         def form_tag_html(html_options)
441           extra_tags = extra_tags_for_form(html_options)
442           tag(:form, html_options, true) + extra_tags
443         end
444         
445         def form_tag_in_block(html_options, &block)
446           content = capture(&block)
447           concat(form_tag_html(html_options), block.binding)
448           concat(content, block.binding)
449           concat("</form>", block.binding)
450         end
