486     # A helper to make it easier to test different route configurations.
487     # This method temporarily replaces ActionController::Routing::Routes
488     # with a new RouteSet instance. 
489     #
490     # The new instance is yielded to the passed block. Typically the block
491     # will create some routes using <tt>map.draw { map.connect ... }</tt>:
492     #
493     #   with_routing do |set|
494     #     set.draw do |map|
495     #       map.connect ':controller/:action/:id'
496     #         assert_equal(
497     #           ['/content/10/show', {}],
498     #           map.generate(:controller => 'content', :id => 10, :action => 'show')
499     #       end
500     #     end
501     #   end
502     #
503     def with_routing
504       real_routes = ActionController::Routing::Routes
505       ActionController::Routing.module_eval { remove_const :Routes }
506 
507       temporary_routes = ActionController::Routing::RouteSet.new
508       ActionController::Routing.module_eval { const_set :Routes, temporary_routes }
509 
510       yield temporary_routes
511     ensure
512       if ActionController::Routing.const_defined? :Routes
513         ActionController::Routing.module_eval { remove_const :Routes }
514       end
515       ActionController::Routing.const_set(:Routes, real_routes) if real_routes
516     end
